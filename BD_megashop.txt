CREATE TABLE CATEGORIA(
ID INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
DESCRICAO VARCHAR(50)
)

CREATE TABLE PRODUTO(
ID INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
ID_CATEGORIA INTEGER NOT NULL FOREIGN KEY REFERENCES CATEGORIA(ID),
NOME VARCHAR(50),
DESCRICAO VARCHAR(500),
VALOR FLOAT
)

CREATE TABLE USUARIO(
ID INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
NOME VARCHAR(50),
CPF VARCHAR(11),
RG VARCHAR(9),
ENDERECO VARCHAR(200),
BAIRRO VARCHAR(50),
CIDADE VARCHAR(50),
CEP VARCHAR(8),
UF VARCHAR(2),
REFERENCIA VARCHAR(50),
TELEFONE VARCHAR(11),
CELULAR VARCHAR(12),
NUM_CARTAO VARCHAR(16),
BANDEIRA VARCHAR(30)
)

CREATE TABLE COMPRA(
ID INTEGER NOT NULL PRIMARY KEY GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
ID_PRODUTO INTEGER NOT NULL FOREIGN KEY REFERENCES PRODUTO(ID),
ID_USUARIO INTEGER NOT NULL FOREIGN KEY REFERENCES USUARIO(ID)
)
